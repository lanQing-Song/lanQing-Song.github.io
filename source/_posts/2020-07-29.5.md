        1.内存溢出
            *一种程序运行出现的错误
            *当程序运行需要的内存超过了剩余的内存时，就会抛出内存溢出错误
        2.内存泄漏
            *占用的无用内存没有及时释放导致内存泄漏，内存泄漏积累多了就容易
                导致内存溢出
        3.常见的内存泄漏
            *意外的全局变量
            *没有及时清理的定时器或回调函数
            *闭包
     -->

     <script>
        //  1.内存溢出
            var obj = {}
            for(var i=0;i<100;i++){
                obj[i] = new Array(100)//过大会导致溢出
                console.log("jiojoj")
            }

            // 2内存泄漏
                // 意外的全局变量
            function fn(){
                a = new Array(10)
                console.log(a)
            }
            fn()   //没有var a 声明导致函数调用完a没有被释放

            // 2内存泄漏
                // 没有清理定时器
            var clean=setInterval(() => {
                console.log("hahah")
            }, 1000);
            clearInterval(clean)//
            //要用变量来接收定时器，再清理

            // 2内存泄漏
            // 没有释放闭包
            function bibao(){
                var a = 2
                function bib(){
                    a++
                    console.log(a)
                }
                return bib
            }
            var bibao2 = bibao()
            bibao2()
            bibao2 = null  //要用null来释放闭包  
            // bibao2()
	
	
			 //  面试题1：
            var name = "The Window";
            var  object = {
                name:"My Object" ,
                getNameFunc : function(){
                    return function(){
                        return this.name;
                    };
                }
            };
            alert(object.getNameFunc()());    //The Window  (调用者是window，window在全局中找到name属性)


        //面试题2：
            var name2 = "The Window";
            var object2 ={
                name2:"My Object",
                getNameFunc:function(){
                    var that = this;
                    return function(){
                        return that.name2;
                    };
                }
            };
            alert(object2.getNameFunc()());  //My Object  此处产生闭包
       console.log("------------------------------------------------------")
        //面试题3：
            function fun(n,o){
                console.log(o)
                return {
                    fun:function(m){
                        return fun(m,n) //
                    }
                }
            }
            var a = fun(0)  //undefined 只产生了一个闭包n=0，所以值不变，都是0
            a.fun(1)  //0
            a.fun(2)  //0
            a.fun(3)  //0

            var b = fun(0).fun(1).fun(2).fun(3)  //undefined  0 1 2 

            var c = fun(0).fun(1)  //undefined 0
            c.fun(2)  //1
            c.fun(3)  //1
     </script>
