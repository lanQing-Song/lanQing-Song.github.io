```javascript
    <!-- 

        1.什么是对象？
            *多个数据的封装体
            *用来保存多个数据的容器
            *一个对象代表现实中的一个事物

        2.为什么要用对象
            *统一管理多个数据
            
        3.对象的组成
            *属性：属性由属性名(本质是字符串）和属性值(可以是任意类型）组成
            *方法：一种特殊的属性（属性值为函数的叫方法）
        
        4.如何访问对象内部数据
            *实例.属性名    (编码简单，有时不能用）
            *实例["属性名"]  （编码复杂，能通用）


        5.什么时候必须使用["属性名"]的方式
            1.属性名包含特殊字符：-、空格
            2.属性名的值是个变量时（属性名不确定）

     -->

     <script>
         var p = {
             "name":"Tom",
             age:12,
             setName:function(name){
                 this.name = name;
             },
             setAge: function(age){
                
                this.age = age;
             }
         }

         p.setName("Bob");
         p["setAge"](23);
         console.log(p.name,p['age']);


         var q = {};
         //1.给q对象添加一个属性：content type： text/json 不能用
         //2.p.content-type = "text/json"  不能用

         p["content-type"]="text/json";   //可以
         console.log(p["content-type"]);  // 可以


         //2.属性名不确定
         var propName = "myAge";
         var value = 18;
         //p.propName =value;  //不能用
         p[propName]=value;    //可以
         console.log(p[propName]);  //可以


     </script>
</body>
</html>
```
