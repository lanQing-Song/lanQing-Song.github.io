<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document</title>
    <script >
        // 1.第一种闭包封装模式
        // 向外暴露一个方法
        // function myModule(){
        //     var msg = "wahaha"
        //     function doSomething(){
        //         console.log("doSomething()：" + msg.toUpperCase())  //toUpperCase()转为大写字母

        //     }
        //     function doOtherthing(){
        //         console.log("doOtherthing()：" + msg.toLowerCase()) //转换为小写字母
        //     }

        //     return doSomething
        // }

        // 2.返回对象，向外暴露多个方法
        function myModule(){
            var msg = "wahaha"
            function doSomething(){
                console.log("doSomething()：" + msg.toUpperCase())  //toUpperCase()转为大写字母

            }
            function doOtherthing(){
                console.log("doOtherthing()：" + msg.toLowerCase()) //转换为小写字母
            }

            return {
                doSomething: doSomething ,
                doOtherthing: doOtherthing
            }
        }


        //3.采用匿名函数自调用方法
        (function(window){
            var msg = "wahaha"
            function doSomething(){
                console.log("doSomething()：" + msg.toUpperCase())  //toUpperCase()转为大写字母

            }
            function doOtherthing(){
                console.log("doOtherthing()：" + msg.toLowerCase()) //转换为小写字母
            }

            //将要暴露的东西添加为window的属性，这种方法在外面可以直接使用,不用调用函数
            window.module2 = {
                doSomething : doSomething ,
                doOtherthing : doOtherthing
            }
        })(window)
    </script>
</head>
<body>
    <!-- 
        闭包的应用：定义js模块
			js模块：
            *具体特定功能的js文件
            *将所有数据和功能都封装在一个函数内部（只有函数内部外界是访问不到的）
            *只向外暴露一个包含n个方法的对象或函数
            *模块的使用者，只需要通过模块暴露的对象调用方法来实现对应的功能
     -->

     <!-- <script src="./myModule1.js"></script>
     <script src="./myModule2.js"></script> -->
     <script>
                //暴露一种方法
                // var fun = myModule()
                // fun()

                // var module =myModule()
                // module.doSomething()
                // module.doOtherthing()


                module2.doSomething()
                module2.doOtherthing()
     </script>
    
</body>
</html>
