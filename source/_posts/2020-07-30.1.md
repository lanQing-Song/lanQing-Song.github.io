     <!-- 
        方式二：对象字面量模式
            *套路：适用{}常见对象，同时指定属性/方法
            *适用场景：起始时对象内部数据是确定的
            *问题：如果创建对个对象，有重复代码

     -->
     <script>
         var p = {
             name:"Tom",
             age:23,
             setNmae:function(name){
                 this.name=name
             }
         }
         //测试
            console.log(p.name,p.age)
            p.setNmae("sb")
            console.log(p.name,p.age)

            var p2 = {
             name:"Tom",
             age:23,
             setNmae:function(name){
                 this.name=name
             }
         }
         console.log(p2.name,p2.age)
         p2.setNmae("sb")
         console.log(p2.name,p2.age)

     </script>
